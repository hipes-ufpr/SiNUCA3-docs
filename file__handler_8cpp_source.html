82int sinucaTracer::TraceFileWriter::AppendToBuffer(void *ptr,
83 unsigned long len) {
84 if (BUFFER_SIZE - this->tf.offsetInBytes < len) {
85 return 1;
86 }
87 memcpy(this->tf.buf + this->tf.offsetInBytes, ptr, len);

104unsigned long sinucaTracer::GetPathTidInSize(const char *sourceDir,
105 const char *prefix,
106 const char *imageName) {
107 unsigned long sourceDirLen = strlen(sourceDir);
108 unsigned long prefixLen = strlen(prefix);
109 unsigned long imageNameLen = strlen(imageName);
110 // 10 is the number of characters on the format string
111 return MAX_INT_DIGITS + 10 + sourceDirLen + prefixLen + imageNameLen;
