74template <typename Type>
75int Queue<Type>::SetConfigParameter(const char* parameter, ConfigValue value) {
76 if (strcmp(parameter, "sendTo") == 0) {
77 if (value.type == ConfigValueTypeComponentReference) {
78 Linkable* linkable = value.value.componentReference;
79 Component<Type>* component =
80 dynamic_cast<Component<Type>*>(linkable);
81 if (component != NULL) {
82 this->sendTo = component;
83 return 0;
84 }
85
86 SINUCA3_ERROR_PRINTF(
87 "Queue parameter is not a component of the queue type.\n");
88 }
89
90 SINUCA3_ERROR_PRINTF(
91 "Queue parameter sendTo is not a component pointer.\n");
92 return 1;
93 } else if (strcmp(parameter, "throughput") == 0) {
94 if (value.type == ConfigValueTypeInteger) {
95 this->throughput = value.value.integer;
96 }
97
98 return 0;
99 }
100
101 SINUCA3_ERROR_PRINTF("Queue received unknown parameter %s.\n", parameter);
102
103 return 1;
104}

106template <typename Type>
107void Queue<Type>::Clock() {
108 long numberOfConnections = this->GetNumberOfConnections();
109 Type packet;
110
111 for (long i = 0; i < numberOfConnections; ++i) {
112 while (this->ReceiveRequestFromConnection(i, &packet) == 0) {
113 if (this->sendTo->SendRequest(this->connectionID, &packet) != 0) {
114 return;
115 }
116 }
117 }
118}
