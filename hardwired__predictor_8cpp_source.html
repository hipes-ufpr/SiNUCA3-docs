50int HardwiredPredictor::SetConfigParameter(const char* parameter,
51 ConfigValue value) {
52 if (strcmp("sendTo", parameter) == 0) {
53 if (value.type == ConfigValueTypeComponentReference) {
54 this->sendTo = dynamic_cast<Component<PredictorPacket>*>(
55 value.value.componentReference);
56 if (this->sendTo != NULL) return 0;
57 }
58 SINUCA3_ERROR_PRINTF(
59 "HardwiredPredictor parameter sendTo is not a "
60 "Component<PredictorPacket>.\n");
61 return 1;
62 }
63
64 if (strcmp("syscall", parameter) == 0)
65 return this->SetBoolParameter(parameter, &this->syscall, value);
66 else if (strcmp("call", parameter) == 0)
67 return this->SetBoolParameter(parameter, &this->call, value);
68 else if (strcmp("return", parameter) == 0)
69 return this->SetBoolParameter(parameter, &this->ret, value);
70 else if (strcmp("uncond", parameter) == 0)
71 return this->SetBoolParameter(parameter, &this->uncond, value);
72 else if (strcmp("cond", parameter) == 0)
73 return this->SetBoolParameter(parameter, &this->cond, value);
74 else if (strcmp("noBranch", parameter) == 0)
75 return this->SetBoolParameter(parameter, &this->noBranch, value);
76
77 SINUCA3_ERROR_PRINTF("HardwiredPredictor received unknown parameter %s.\n",
78 parameter);
79 return 1;
80}

82void HardwiredPredictor::Respond(int id, PredictorPacket request) {
83 if (request.type == PredictorPacketTypeRequestUpdate) return;
84 const InstructionPacket instruction = request.data.requestQuery;
85 bool predict = true;
86
87 if (!instruction.staticInfo->isControlFlow) {
88 ++this->numberOfNoBranchs;
89 predict = this->noBranch;
90 } else {
91 switch (instruction.staticInfo->branchType) {
92 case BranchSyscall:
93 predict = this->syscall;
94 ++this->numberOfSyscalls;
95 break;
96 case BranchCall:
97 predict = this->call;
98 ++this->numberOfCalls;
99 break;
100 case BranchReturn:
101 predict = this->ret;
102 ++this->numberOfRets;
103 break;
104 case BranchUncond:
105 predict = this->uncond;
106 ++this->numberOfUnconds;
107 break;
108 case BranchCond:
109 predict = this->cond;
110 ++this->numberOfConds;
111 break;
112 }
113 }
114
115 PredictorPacket response;
116 response.type = PredictorPacketTypeResponseTakeToAddress;
117 response.data.response.instruction = instruction;
118 if (predict) {
119 response.data.response.target = instruction.nextInstruction;
120 } else {
121 // Fast way to create an address that isn't nextInstruction.
122 response.data.response.target = ~instruction.nextInstruction;
123 }
124
125 if (this->sendTo != NULL) {
126 this->sendTo->SendRequest(this->sendToID, &response);
127 } else {
128 this->SendResponseToConnection(id, &response);
129 }
130}

132void HardwiredPredictor::Clock() {
133 const unsigned long numberOfConnections = this->GetNumberOfConnections();
134 for (unsigned long i = 0; i < numberOfConnections; ++i) {
135 PredictorPacket packet;
136 while (this->ReceiveRequestFromConnection(i, &packet) == 0) {
137 this->Respond(i, packet);
138 }
139 }
