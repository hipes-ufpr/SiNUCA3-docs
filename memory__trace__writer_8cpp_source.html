52void sinucaTracer::MemoryTraceFile::PrepareDataNonStdAccess(
53 PIN_MULTI_MEM_ACCESS_INFO* pinNonStdInfo) {
54 this->numReadOps = 0;
55 this->numWriteOps = 0;
56
57 for (unsigned int it = 0; it < pinNonStdInfo->numberOfMemops; ++it) {
58 if (pinNonStdInfo->memop[it].memopType == PIN_MEMOP_LOAD) {
59 this->readOps[this->numReadOps].addr =
60 pinNonStdInfo->memop[it].memoryAddress;
61 this->readOps[this->numReadOps].size =
62 pinNonStdInfo->memop[it].bytesAccessed;
63 this->numReadOps++;
64 } else {
65 this->readOps[this->numReadOps].addr =
66 pinNonStdInfo->memop[it].memoryAddress;
67 this->writeOps[this->numWriteOps].size =
68 pinNonStdInfo->memop[it].bytesAccessed;
69 this->numWriteOps++;
70 }
71 }
72 // This variable is checked in AppendToBufferLastMemoryAccess
73 this->wasLastOperationStd = false;
74}

84void sinucaTracer::MemoryTraceFile::AppendToBufferLastMemoryAccess() {
85 if (this->wasLastOperationStd) {
86 this->MemoryAppendToBuffer(&this->stdAccessOp,
87 sizeof(this->stdAccessOp));
88 } else {
89 // Append number of read operations
90 this->MemoryAppendToBuffer(&this->numReadOps, SIZE_NUM_MEM_R_W);
91 // Append number of write operations
92 this->MemoryAppendToBuffer(&this->numWriteOps, SIZE_NUM_MEM_R_W);
93 // Append read operations' buffer
94 this->MemoryAppendToBuffer(this->readOps,
95 this->numReadOps * sizeof(*this->readOps));
96 // Append write operations' buffer
97 this->MemoryAppendToBuffer(this->writeOps,
98 this->numWriteOps * sizeof(*this->writeOps));
99 }
100}

103 size_t len) {
