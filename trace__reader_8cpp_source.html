28int sinucaTracer::SinucaTraceReader::OpenTrace(const char *imageName,
29 const char *sourceDir) {
30 StaticTraceFile staticFile(sourceDir, imageName);
31 if (!staticFile.Valid()) return 1;
32
33 this->totalThreads = staticFile.GetNumThreads();
34 this->binaryTotalBBLs = staticFile.GetTotalBBLs();
35 this->thrsInfo = new ThrInfo[this->totalThreads];
36
37 for (int i = 0; i < this->totalThreads; i++) {
38 if (this->thrsInfo[i].Allocate(sourceDir, imageName, i)) {
39 return 1;
40 }
41 }
42
43 if (this->GenerateBinaryDict(&staticFile)) {
44 SINUCA3_ERROR_PRINTF("Failed to generate instruction dictionary\n");
45 return 1;
46 }
47
48 return 0;
49}

58int sinucaTracer::SinucaTraceReader::GenerateBinaryDict(
59 StaticTraceFile *stFile) {
60 InstructionInfo *instInfoPtr;
61 unsigned long poolOffset;
62 unsigned int bblSize;
63 unsigned int instCounter;
64 unsigned int bblCounter;
65
66 this->binaryBBLsSize = new unsigned int[this->binaryTotalBBLs];
67 this->binaryDict = new InstructionInfo *[this->binaryTotalBBLs];
68 this->pool = new InstructionInfo[stFile->GetTotalIns()];
69 poolOffset = 0;
70
71 for (bblCounter = 0; bblCounter < this->binaryTotalBBLs; bblCounter++) {
72 if (stFile->GetNewBBLSize(&bblSize)) {
73 return 1;
74 }
75 this->binaryBBLsSize[bblCounter] = bblSize;
76 this->binaryDict[bblCounter] = &this->pool[poolOffset];
77 poolOffset += bblSize;
78
79 for (instCounter = 0; instCounter < bblSize; instCounter++) {
80 instInfoPtr = &this->binaryDict[bblCounter][instCounter];
81 stFile->ReadNextInstruction(instInfoPtr);
82 }
83
84 SINUCA3_DEBUG_PRINTF("bbl [%u] size [%u]\n", bblCounter + 1, bblSize);
85 }
86
87 return 0;
88}

90FetchResult sinucaTracer::SinucaTraceReader::Fetch(InstructionPacket *ret,
91 int tid) {
92 InstructionInfo *packageInfo;
93
94 if (!this->thrsInfo[tid].isInsideBBL) {
95 if (this->thrsInfo[tid].dynFile->ReadNextBBl(
96 &this->thrsInfo[tid].currentBBL)) {
97 return FetchResultEnd;
98 }
99 this->thrsInfo[tid].isInsideBBL = true;
100 this->thrsInfo[tid].currentOpcode = 0;
101 }
102
103 unsigned int currentBbl = this->thrsInfo[tid].currentBBL;
104 unsigned int currentIns = this->thrsInfo[tid].currentOpcode;
105 packageInfo = &this->binaryDict[currentBbl][currentIns];
106 ret->staticInfo = &(packageInfo->staticInfo);
107 this->thrsInfo[tid].memFile->ReadNextMemAccess(packageInfo,
108 &ret->dynamicInfo);
109
110 this->thrsInfo[tid].currentOpcode++;
111 if (this->thrsInfo[tid].currentOpcode >=
112 this->binaryBBLsSize[this->thrsInfo[tid].currentBBL]) {
113 this->thrsInfo[tid].isInsideBBL = false;
114 }
115
116 this->thrsInfo[tid].fetchedInst++;
117
118 SINUCA3_DEBUG_PRINTF("Fetched: %s\n", (*ret).staticInfo->opcodeAssembly);
119
120 return FetchResultOk; // Maybe this is not suitable for multiple threads
121}

152int sinucaTracer::ThrInfo::Allocate(const char *sourceDir,
153 const char *imageName, int tid) {

155 if (!this->dynFile->Valid()) return 1;
