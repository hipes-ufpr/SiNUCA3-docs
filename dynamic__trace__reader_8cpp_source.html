33sinucaTracer::DynamicTraceFile::DynamicTraceFile(const char *folderPath,
34 const char *imageName,
35 THREADID tid) {
36 unsigned long bufferSize =
37 GetPathTidInSize(folderPath, "dynamic", imageName);
38 char *path = (char *)alloca(bufferSize);
39
40 FormatPathTidIn(path, folderPath, "dynamic", imageName, tid, bufferSize);
41 if (this->UseFile(path) == NULL) {
42 this->isValid = false;
43 return;
44 }
45
46 /*
47 * The number of executed instructions is placed at the top of the dynamic
48 * file.
49 */
50 fread(&this->totalExecInst, sizeof(this->totalExecInst), 1, this->tf.file);
51 SINUCA3_DEBUG_PRINTF("totalExecInst [%lu]\n", this->totalExecInst);
52
53 this->bufActiveSize =
54 (unsigned int)(BUFFER_SIZE / sizeof(BBLID)) * sizeof(BBLID);
55 this->RetrieveBuffer(); /* First buffer read */
56 this->isValid = true;
57}
