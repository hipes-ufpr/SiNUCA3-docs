37void SimpleExecutionUnit::Clock() {
38 unsigned long numberOfConnections = this->GetNumberOfConnections();
39 for (unsigned long i = 0; i < numberOfConnections; ++i) {
40 InstructionPacket packet;
41 while (this->ReceiveRequestFromConnection(i, &packet) == 0) {
42 SINUCA3_DEBUG_PRINTF("[SimpleExecutionUnit] %p: executing %s.\n",
43 this, packet.staticInfo->opcodeAssembly);
44 ++this->numberOfInstructions;
45 }
46 }
47}
