34sinucaTracer::StaticTraceFile::StaticTraceFile(const char *folderPath,
35 const char *img) {
36 unsigned long bufferLen = GetPathTidOutSize(folderPath, "static", img);
37 char *staticPath = (char *)alloca(bufferLen);
38 FormatPathTidOut(staticPath, folderPath, "static", img, bufferLen);
39
40 this->fd = open(staticPath, O_RDONLY);
41 if (fd == -1) {
42 printFileErrorLog(staticPath, "O_RDONLY");
43 this->isValid = false;
44 return;
45 }
46 this->mmapSize = lseek(fd, 0, SEEK_END);
47 SINUCA3_DEBUG_PRINTF("Mmap Size [%lu]\n", this->mmapSize);
48 this->mmapPtr =
49 (char *)mmap(0, this->mmapSize, PROT_READ, MAP_PRIVATE, fd, 0);
50 if (this->mmapPtr == MAP_FAILED) {
51 printFileErrorLog(staticPath, "PROT_READ MAP_PRIVATE");
52 this->isValid = false;
53 return;
54 }
55
56 /* Skip static trace header information */
57 this->mmapOffset = 3 * sizeof(unsigned int);
58
59 this->totalThreads = *(unsigned int *)(this->mmapPtr + 0);
60 SINUCA3_DEBUG_PRINTF("Number of Threads [%u]\n", this->totalThreads);
61 this->totalBBLs = *(unsigned int *)(this->mmapPtr + sizeof(unsigned int));
62 SINUCA3_DEBUG_PRINTF("Number of BBLs [%u]\n", this->totalBBLs);
63 this->totalIns =
64 *(unsigned int *)(this->mmapPtr + 2 * sizeof(unsigned int));
65 SINUCA3_DEBUG_PRINTF("Number of Instructions [%u]\n", this->totalIns);
66
67 this->isValid = true;
68 this->instLeftInBBL = 0;
69}

71void sinucaTracer::StaticTraceFile::ReadNextInstruction(InstructionInfo *info) {
72 DataINS *data = (DataINS *)(this->GetData(sizeof(DataINS)));
73
74 unsigned long len = strlen(data->name);
75 memcpy(info->staticInfo.opcodeAssembly, data->name, len + 1);
76
77 info->staticInfo.opcodeSize = data->size;
78 info->staticInfo.baseReg = data->baseReg;
79 info->staticInfo.indexReg = data->indexReg;
80 info->staticInfo.opcodeAddress = data->addr;
81
82 this->GetBooleanValues(&info->staticInfo, data);
83 this->GetBranchType(&info->staticInfo, data);
84 this->GetRegisters(&info->staticInfo, data);
85
86 if (!info->staticInfo.isNonStdMemOp) {
87 info->staticNumReadings = data->isRead + data->isRead2;
88 info->staticNumWritings = data->isWrite;
89 }
90
91 this->instLeftInBBL--;
92}

116void sinucaTracer::StaticTraceFile::GetBranchType(StaticInstructionInfo *info,
117 DataINS *data) {
118 switch (data->branchType) {
119 case BRANCH_CALL:
120 info->branchType = BranchCall;
121 break;
122 case BRANCH_SYSCALL:
123 info->branchType = BranchSyscall;
124 break;
125 case BRANCH_RETURN:
126 info->branchType = BranchReturn;
127 break;
128 case BRANCH_COND:
129 info->branchType = BranchCond;
130 break;
131 case BRANCH_UNCOND:
132 info->branchType = BranchUncond;
133 break;

137void sinucaTracer::StaticTraceFile::GetRegisters(StaticInstructionInfo *info,
138 struct DataINS *data) {
139 info->numReadRegs = data->numReadRegs;
140 memcpy(info->readRegs, data->readRegs,
141 data->numReadRegs * sizeof(*data->readRegs));
142
143 info->numWriteRegs = data->numWriteRegs;
144 memcpy(info->writeRegs, data->writeRegs,
145 data->numWriteRegs * sizeof(*data->writeRegs));
146}
