142int EngineBuilder::Yaml2Parameter(const char* name, builder::Parameter* dest,
143 const yaml::YamlValue* src) {
144 int err;
145
146 switch (src->type) {
147 case yaml::YamlValueTypeBoolean:
148 dest->type = builder::ParameterTypeBoolean;
149 dest->value.boolean = src->value.boolean;
150 break;
151 case yaml::YamlValueTypeNumber:
152 YamlNumber2Parameter(dest, src->value.number);
153 break;
154 case yaml::YamlValueTypeArray:
155 YamlArray2Parameter(dest, src->value.array);
156 break;
157 case yaml::YamlValueTypeMapping:
158 dest->type = builder::ParameterTypeDefinitionReference;
159 err = this->AddComponentDefinitionFromYamlMapping(
160 name, src->value.mapping, &dest->value.referenceToDefinition);
161 if (err != 0) return 1;
162 break;
163 case yaml::YamlValueTypeString:
164 dest->value.referenceToDefinition =
165 this->GetComponentDefinitionOrMakeDummy(src->value.string);
166 dest->type = builder::ParameterTypeDefinitionReference;
167 break;
168 case yaml::YamlValueTypeAlias:
169 if (strcmp(src->value.alias, "ENGINE") == 0) {
170 dest->value.referenceToInstance = 0;
171 } else {
172 dest->value.referenceToInstance =
173 this->GetComponentInstantiationOrMakeDummy(
174 src->value.alias);
175 }
176 dest->type = builder::ParameterTypeInstanceReference;
177 break;
178 }
179
180 return 0;
181}

206int EngineBuilder::FillParametersAndClass(
207 builder::DefinitionID id,
208 const std::vector<yaml::YamlMappingEntry*>* config) {
209 SINUCA3_DEBUG_PRINTF(" Filling parameters and class for definition.\n");
210
211 builder::ComponentDefinition* definition = &this->componentDefinitions[id];
212
213 definition->clazz = NULL;
214
215 if (config->size() == 0) {
216 SINUCA3_ERROR_PRINTF(
217 "While trying to define component %s: parameter `class` not "
218 "provided.\n",
219 definition->name);
220 return 1;
221 }
222
223 // It's -1 because we don't append the class.
224 definition->parameters.size = config->size() - 1;
225 definition->parameters.items =
226 new builder::ParameterMapItem[definition->parameters.size];
227
228 unsigned long itemsIndex = 0;
229 for (unsigned long i = 0; i < config->size(); ++i) {
230 // Get the class.
231 if (strcmp((*config)[i]->name, "class") == 0) {
232 if (AddClass(definition, (*config)[i]->value)) return 1;
233
234 continue;
235 }
236
237 // If we're on the last item and class was not provided, appending would
238 // overflow, but we know that class was not provided!
239 if (i == config->size() && definition->clazz == NULL) {
240 SINUCA3_ERROR_PRINTF(
241 "While trying to define component %s: parameter `class` was "
242 "not provided.\n",
243 definition->name);
244 return 1;
245 }
246
247 // Finally add the parameter.
248 builder::ParameterMapItem* parameter =
249 &definition->parameters.items[itemsIndex];
250 parameter->name = (*config)[i]->name;
251 if (this->Yaml2Parameter(parameter->name, &parameter->value,
252 (*config)[i]->value))
253 return 1;
254
255 ++itemsIndex;
256 }
257
258 return 0;
259}

261int EngineBuilder::AddComponentDefinitionFromYamlMapping(
262 const char* name, const std::vector<yaml::YamlMappingEntry*>* config,
263 builder::DefinitionID* ret) {
264 builder::DefinitionID definitionID;
265 bool mustPop = false;
266
267 if (this->GetComponentDefinition(name, &definitionID)) {
268 builder::ComponentDefinition definition =
269 builder::ComponentDefinition(name, true);
270 this->componentDefinitions.push_back(definition);
271
272 definitionID = this->componentDefinitions.size() - 1;
273 mustPop = true;
274 } else if (this->componentDefinitions[definitionID].alreadyDefined) {
275 SINUCA3_ERROR_PRINTF("Multiple definitions of component %s.", name);
276 return 1;
277 }
278
279 if (this->FillParametersAndClass(definitionID, config)) {
280 if (mustPop) this->componentDefinitions.pop_back();
281 return 1;
282 }
283
284 *ret = definitionID;
285 return 0;
286}

314

318 return 0;

347int EngineBuilder::TreatParameter(const char* name,
348 const yaml::YamlValue* value) {
349 SINUCA3_DEBUG_PRINTF("Treating parameter %s.\n", name);
350
351 if (strcmp(name, "instantiate") == 0) {
352 return this->TreatInstantiateParameter(value);
353 }
354
355 // If the parameter is not an "instantiate" special parameter, it's a
356 // component definition.
357 SINUCA3_DEBUG_PRINTF("Parameter is a component definition.\n");
358 if (value->type != yaml::YamlValueTypeMapping) {
359 SINUCA3_ERROR_PRINTF(
360 "While trying to define component %s: expected a YAML Mapping, "
361 "got %s.\n",
362 name, value->TypeAsString());
363 return 1;
364 }
365
366 SINUCA3_DEBUG_PRINTF(" With anchor %s.\n", value->anchor);
367 builder::InstanceID dummy;
368 if (value->anchor != NULL) {
369 if (this->AddComponentInstantiationFromYamlMapping(
370 name, value->anchor, value->value.mapping, &dummy) != 0)
371 return 1;
372
373 SINUCA3_DEBUG_PRINTF(" Successfully added component instantiation.\n");
374 return 0;
375 }
376
377 if (this->AddComponentDefinitionFromYamlMapping(name, value->value.mapping,
378 &dummy) != 0)
379 return 1;
380
381 SINUCA3_DEBUG_PRINTF(" Successfully added component definition.\n");
382 return 0;
383}

470int EngineBuilder::Parameter2ConfigValue(const builder::Parameter* parameter,
471 ConfigValue* value) {
472 switch (parameter->type) {
473 case builder::ParameterTypeInteger:
474 *value = ConfigValue(parameter->value.integer);
475 break;
476 case builder::ParameterTypeNumber:
477 *value = ConfigValue(parameter->value.number);
478 break;
479 case builder::ParameterTypeBoolean:
480 *value = ConfigValue(parameter->value.boolean);
481 break;
482 case builder::ParameterTypeInstanceReference:
483 *value = ConfigValue(
484 this->components[parameter->value.referenceToInstance]
485 .component);
486 break;
487 case builder::ParameterTypeArray:
488 return this->ArrayParameter2ConfigValue(&parameter->value.array,
489 value);
490 break;
491 case builder::ParameterTypeDefinitionReference:
492 *value = ConfigValue(this->NewComponentFromDefinitionReference(
493 parameter->value.referenceToDefinition));
494 if (value->value.componentReference == NULL) return 1;
495 break;
496 default:
497 assert(0 && "unreachable");
498 }
499
500 return 0;
501}

503int EngineBuilder::SetupComponentConfig(builder::InstanceID instance) {
504 Linkable* component = this->components[instance].component;
505 builder::DefinitionID definition = this->components[instance].definition;
506
507 // We need to perform the access to map each iteration as
508 // Parameter2ConfigValue may append to the array, thus changing the location
509 // of the map.
510 for (long i = 0; i < this->componentDefinitions[definition].parameters.size;
511 ++i) {
512 builder::ParameterMap* map =
513 &this->componentDefinitions[definition].parameters;
514 const char* parameterName = map->items[i].name;
515 const builder::Parameter* parameterValue = &map->items[i].value;
516 ConfigValue value;
517 if (this->Parameter2ConfigValue(parameterValue, &value)) return 1;
518 if (component->SetConfigParameter(parameterName, value)) return 1;
519 }
520
521 return 0;
522}

524Engine* EngineBuilder::Instantiate(const char* configFile) {
525 const yaml::YamlValue* yamlConfig = yaml::ParseFile(configFile);
526 if (yamlConfig == NULL) return NULL;
527 assert(yamlConfig->type == yaml::YamlValueTypeMapping);
528
529 // We only need this to instantiate everything.
530 const std::vector<yaml::YamlMappingEntry*>* config =
531 yamlConfig->value.mapping;
532
533 for (unsigned int i = 0; i < config->size(); ++i) {
534 if (this->TreatParameter((*config)[i]->name, (*config)[i]->value))
535 return this->FreeSelfOnInstantiationFailure(yamlConfig);
536 }
537
538 // Error.
539 if (this->EnsureAllComponentsAreDefined())
540 return this->FreeSelfOnInstantiationFailure(yamlConfig);
541
542 // We instantiate all components first so the pointers to aliased stuff will
543 // always work. We only have to solve definitions references now.
544 //
545 // We skip the engine of course.
546 for (unsigned long i = 1; i < this->components.size(); ++i) {
547 builder::ComponentInstantiation* component = &this->components[i];
548 builder::ComponentDefinition* definition =
549 &this->componentDefinitions[component->definition];
550 component->component = CreateComponent(definition->clazz);
551
552 // No such class.
553 if (component->component == NULL) {
554 SINUCA3_ERROR_PRINTF("No such component class: %s.\n",
555 definition->clazz);
556 return this->FreeSelfOnInstantiationFailure(yamlConfig);
557 }
558 }
559
560 // Pass the parameters to the components. This also resolves pointers to
561 // definitions, thus allocating more components. As we already ensured all
562 // definitions are satisfied, this cannot fail.
563 //
564 // We skip the engine of course.
565 for (unsigned long i = 1; i < this->components.size(); ++i) {
566 if (this->SetupComponentConfig(i))
567 return this->FreeSelfOnInstantiationFailure(yamlConfig);
568 }
569 for (unsigned long i = 1; i < this->components.size(); ++i) {
570 if (this->components[i].component->FinishSetup()) {
571 return this->FreeSelfOnInstantiationFailure(yamlConfig);
572 }
573 }
574
575 Engine* engine = this->BuildEngine();
576 delete yamlConfig;
577 return engine;
578}

580Engine* EngineBuilder::BuildEngine() {
581 long numberOfComponents = this->components.size();
582 Linkable** components = new Linkable*[numberOfComponents];
583
584 assert(components != NULL);
