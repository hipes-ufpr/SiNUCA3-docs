30int BTBEntry::Allocate(unsigned int numBanks) {
31 this->numBanks = numBanks;
32 this->targetArray = new unsigned long[numBanks];
33 if (!(this->targetArray)) {
34 SINUCA3_ERROR_PRINTF("BTB entry could not be allocated");
35 return 1;
36 }
37
38 this->branchTypes = new BranchType[numBanks];
39 if (!(this->branchTypes)) {
40 delete[] this->targetArray;
41 SINUCA3_ERROR_PRINTF("BTB entry could not be allocated");
42 return 1;
43 }
44
45 this->predictorsArray = new BimodalPredictor[numBanks];
46 if (!(this->predictorsArray)) {
47 delete[] this->targetArray;
48 delete[] this->branchTypes;
49 SINUCA3_ERROR_PRINTF("BTB entry could not be allocated");
50 return 1;
51 }
52
53 for (unsigned int i = 0; i < this->numBanks; ++i) {
54 this->targetArray[i] = 0;
55 this->branchTypes[i] = BranchTypeNone;
56 }
57
58 return 0;
59}

61int BTBEntry::NewEntry(unsigned long tag, unsigned int bank,
62 unsigned long target,
63 const StaticInstructionInfo* instruction) {
64 if (bank >= this->numBanks) return 1;
65
66 this->entryTag = tag;
67 this->targetArray[bank] = target;
68 this->branchTypes[bank] =
69 BranchTypeFromSinucaBranch(instruction->branchType);
70
71 return 0;
72}

74int BTBEntry::UpdateEntry(unsigned long bank, bool branchState) {
75 if ((bank >= this->numBanks) || this->entryTag == 0) return 1;
76
77 this->predictorsArray[bank].UpdatePrediction(branchState);
78 return 0;

103int BranchTargetBuffer::SetConfigParameter(const char* parameter,
104 ConfigValue value) {
105 if (strcmp(parameter, "interleavingFactor") == 0) {
106 if (value.type != ConfigValueTypeInteger) {
107 SINUCA3_ERROR_PRINTF(
108 "BTB parameter interleavingFactor is not an integer.\n");
109 return 1;
110 }
111 unsigned int iFactor = value.value.integer;
112 if (iFactor <= 0) {
113 SINUCA3_ERROR_PRINTF(
114 "BTB parameter interleavingFactor must be > 0.\n");
115 return 1;
116 }
117
118 if (iFactor > MAX_INTERLEAVING_FACTOR) {
119 this->interleavingFactor = MAX_INTERLEAVING_FACTOR;
120 } else {
121 this->interleavingFactor = iFactor;
122 }
123
124 return 0;
125 } else if (strcmp(parameter, "numberOfEntries") == 0) {
126 if (value.type != ConfigValueTypeInteger) {
127 SINUCA3_ERROR_PRINTF(
128 "BTB parameter numberOfEntries is not an integer.\n");
129 return 1;
130 }
131 unsigned int entries = value.value.integer;
132 if (entries <= 0) {
133 SINUCA3_ERROR_PRINTF(
134 "BTB parameter numberOfEntries must be > 0.\n");
135 return 1;
136 }
137 this->numEntries = entries;
138
139 return 0;
140 } else if (strcmp(parameter, "sendTo") == 0) {
141 if (value.type != ConfigValueTypeComponentReference) {
142 SINUCA3_ERROR_PRINTF(
143 "BTB parameter sendTo is not a Component<BTBPacket>.\n");
144 return 1;
145 }
146 this->sendTo =
147 dynamic_cast<Component<BTBPacket>*>(value.value.componentReference);
148 if (this->sendTo == NULL) {
149 SINUCA3_ERROR_PRINTF(
150 "BTB parameter sendTo is not a Component<BTBPacket>.\n");
151 return 1;
152 }
153 return 0;
154 }
155
156 SINUCA3_WARNING_PRINTF("BTB received an unknown parameter: %s.\n",
157 parameter);
158 return 1;
159}

161int BranchTargetBuffer::FinishSetup() {
162 if (this->interleavingFactor == 0) {
163 SINUCA3_ERROR_PRINTF(
164 "BTB did not receive the interleaving factor parameter.\n");
165 return 1;
166 }
167
168 if (this->numEntries == 0) {
169 SINUCA3_ERROR_PRINTF(
170 "BTB did not receive the number of entries parameter.\n");
171 return 1;
172 }
173
174 this->interleavingBits = floor(log(this->interleavingFactor));
175 this->entriesBits = floor(log(this->numEntries));
176 this->interleavingFactor = (1 << this->interleavingBits);
177 this->numEntries = (1 << this->entriesBits);
178 this->btb = new BTBEntry*[this->numEntries];
179 if (!(this->btb)) {
180 SINUCA3_ERROR_PRINTF("BTB could not be allocated.\n");
181 return 1;
182 }
183
184 for (unsigned int i = 0; i < this->numEntries; ++i) {
185 this->btb[i] = new BTBEntry;
186 this->btb[i]->Allocate(this->interleavingFactor);
187 }
188
189 if (this->sendTo != NULL) {
190 this->sendToID = this->sendTo->Connect(0);
191 }
192
193 return 0;
194}

236inline void BranchTargetBuffer::Query(const StaticInstructionInfo* instruction,
237 int connectionID) {
238 unsigned long index = this->CalculateIndex(instruction->opcodeAddress);
239 unsigned long tag = this->CalculateTag(instruction->opcodeAddress);
240 BTBPacket response;
241
242 BTBEntry* currentEntry = btb[index];
243 if (currentEntry->GetTag() == tag) {
244 // BTB Hit
245 /*
246 * In a BTB hit, searches for a taken branch instruction and fills in
247 * the target address if it finds one. Marks all instructions before the
248 * taken branch as valid.
249 */
250 bool branchTaken = false;
251 response.data.response.instruction = instruction;
252 response.data.response.target =
253 instruction->opcodeAddress + this->interleavingFactor;
254 response.data.response.numberOfBits = this->interleavingFactor;
255
256 for (unsigned int i = 0; i < this->interleavingFactor; ++i) {
257 if (!(branchTaken)) {
258 if ((currentEntry->GetBranchType(i) ==
259 BranchTypeUnconditional) ||
260 (currentEntry->GetPrediction(i) == TAKEN)) {
261 response.data.response.target =
262 currentEntry->GetTargetAddress(i);
263 branchTaken = true;
264 }
265 response.data.response.validBits[i] = true;
266 } else {
267 response.data.response.validBits[i] = false;
268 }
269 }
270 response.type = BTBPacketTypeResponseBTBHit;
271 } else {
272 // BTB Miss
273 /*
274 * In a BTB Miss, it assumes that all instructions are valid and that
275 * the next fetch block is sequential.
276 */
277 response.data.response.instruction = instruction;
278 response.data.response.target =
279 instruction->opcodeAddress + this->interleavingFactor;
280 response.data.response.numberOfBits = this->interleavingFactor;
281 for (unsigned int i = 0; i < this->interleavingFactor; ++i) {
282 response.data.response.validBits[i] = true;
283 }
284 response.type = BTBPacketTypeResponseBTBMiss;
285 }
286
287 if (this->sendTo == NULL) {
288 this->SendResponseToConnection(connectionID, &response);
289 } else {
290 this->sendTo->SendRequest(this->sendToID, &response);
291 }
292}

304void BranchTargetBuffer::Clock() {
305 long numberOfConnections = this->GetNumberOfConnections();
306 BTBPacket packet;
307 for (long i = 0; i < numberOfConnections; ++i) {
308 if (this->ReceiveRequestFromConnection(i, &packet) == 0) {
309 switch (packet.type) {
310 case BTBPacketTypeRequestQuery:
311 ++this->numQueries;
312 this->Query(packet.data.requestQuery, i);
313 break;
314 case BTBPacketTypeRequestAddEntry:
315 this->AddEntry(packet.data.requestAddEntry.instruction,
316 packet.data.requestAddEntry.target);
317 break;
318 case BTBPacketTypeRequestUpdate:
319 this->Update(packet.data.requestUpdate.instruction,
320 packet.data.requestUpdate.branchState);
321 break;
322 case BTBPacketTypeResponseBTBHit:
323 case BTBPacketTypeResponseBTBMiss:
324 SINUCA3_WARNING_PRINTF(
325 "Connection %ld send a response type message to BTB.\n",
326 i);
327 break;
328 default:
329 SINUCA3_WARNING_PRINTF(
330 "Connection %ld send a invalid message to BTB.\n", i);
331 }
332 }
333 }
334}
