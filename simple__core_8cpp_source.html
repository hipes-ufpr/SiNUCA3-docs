30int SimpleCore::SetConfigParameter(const char* parameter, ConfigValue value) {
31 Component<MemoryPacket>** ptrToParameter;
32 int* connectionIDPtr;
33 if (strcmp(parameter, "instructionMemory") == 0) {
34 ptrToParameter = &this->instructionMemory;
35 connectionIDPtr = &this->instructionConnectionID;
36 } else if (strcmp(parameter, "dataMemory") == 0) {
37 ptrToParameter = &this->dataMemory;
38 connectionIDPtr = &this->dataConnectionID;
39 } else if (strcmp(parameter, "fetching") == 0) {
40 if (value.type != ConfigValueTypeComponentReference) {
41 SINUCA3_ERROR_PRINTF(
42 "Component SimpleCore received a parameter that's not a "
43 "component "
44 "reference.\n");
45 return 1;
46 }
47
48 this->fetching = dynamic_cast<Component<FetchPacket>*>(
49 value.value.componentReference);
50 if (this->fetching == NULL) {
51 SINUCA3_ERROR_PRINTF(
52 "Component SimpleCore received a parameter fetching that's not"
53 "a reference to a Component<MemoryPacket>.\n");
54 return 1;
55 }
56 this->fetchingConnectionID = this->fetching->Connect(0);
57 return 0;
58 } else {
59 SINUCA3_ERROR_PRINTF(
60 "Component SimpleCore received unknown parameter %s.\n", parameter);
61 return 1;
62 }
63
64 if (value.type != ConfigValueTypeComponentReference) {
65 SINUCA3_ERROR_PRINTF(
66 "Component SimpleCore received a parameter that's not a component "
67 "reference.\n");
68 return 1;
69 }
70
71 *ptrToParameter =
72 dynamic_cast<Component<MemoryPacket>*>(value.value.componentReference);
73 if (*ptrToParameter == NULL) {
74 SINUCA3_ERROR_PRINTF(
75 "Component SimpleCore received a parameter %s that's not a "
76 "reference "
77 "to a Component<MemoryPacket>.\n",
78 parameter);
79 return 1;
80 }
81
82 *connectionIDPtr = (*ptrToParameter)->Connect(0);
83
84 return 0;
85}

97void SimpleCore::Clock() {
98 FetchPacket fetch;
99 fetch.request = 0;
100 this->fetching->SendRequest(this->fetchingConnectionID, &fetch);
101 if (this->fetching->ReceiveResponse(this->fetchingConnectionID, &fetch) ==
102 0) {
103 ++this->numFetchedInstructions;
104 if (this->instructionMemory != NULL) {
105 MemoryPacket fetchPacket = fetch.response.staticInfo->opcodeAddress;
106 this->instructionMemory->SendRequest(this->instructionConnectionID,
107 &fetchPacket);
108 if (this->dataMemory != NULL) {
109 for (long i = 0; i < fetch.response.dynamicInfo.numReadings;
110 ++i) {
111 this->dataMemory->SendRequest(
112 this->dataConnectionID,
113 (unsigned long*)&(
114 fetch.response.dynamicInfo.readsAddr[i]));
115 }
116
117 for (long i = 0; i < fetch.response.dynamicInfo.numWritings;
118 ++i) {
119 this->dataMemory->SendRequest(
120 this->dataConnectionID,
121 (unsigned long*)&(
122 fetch.response.dynamicInfo.writesAddr[i]));
123 }
124 }
125 }
126 }
127}
