27int SimpleInstructionMemory::SetConfigParameter(const char* parameter,
28 ConfigValue value) {
29 if (strcmp(parameter, "sendTo") == 0) {
30 if (value.type == ConfigValueTypeComponentReference) {
31 this->sendTo = dynamic_cast<Component<InstructionPacket>*>(
32 value.value.componentReference);
33 if (this->sendTo != NULL) return 0;
34 }
35
36 SINUCA3_ERROR_PRINTF(
37 "SimpleInstructionMemory parameter `sendTo` is not a "
38 "Component<InstructionPacket>.\n");
39 return 1;
40 }
41
42 SINUCA3_ERROR_PRINTF(
43 "SimpleInstructionMemory received unknown parameter %s.\n", parameter);
44 return 1;
45}

54void SimpleInstructionMemory::Clock() {
55 long numberOfConnections = this->GetNumberOfConnections();
56 InstructionPacket packet;
57 for (long i = 0; i < numberOfConnections; ++i) {
58 while (this->ReceiveRequestFromConnection(i, &packet) == 0) {
59 ++this->numberOfRequests;
60 if (this->sendTo != NULL) {
61 this->sendTo->SendRequest(this->sendToID, &packet);
62 } else {
63 this->SendResponseToConnection(i, &packet);
64 }
65 }
66 }
67}
