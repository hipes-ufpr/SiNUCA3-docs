29int Fetcher::FinishSetup() {
30 if (this->fetch == NULL) {
31 SINUCA3_ERROR_PRINTF(
32 "Fetcher didn't received required parameter `fetch`.\n");
33 return 1;
34 }
35 if (this->instructionMemory == NULL) {
36 SINUCA3_ERROR_PRINTF(
37 "Fetcher didn't received required parameter "
38 "`instructionMemory`.\n");
39 return 1;
40 }
41
42 this->fetchID = this->fetch->Connect(this->fetchSize);
43 this->instructionMemoryID =
44 this->instructionMemory->Connect(this->fetchSize);
45
46 this->fetchBuffer = new FetchBufferEntry[this->fetchSize];
47
48 // Maybe connect to a predictor.
49 if (this->predictor != NULL) {
50 this->predictorID = this->predictor->Connect(this->fetchSize);
51 this->flagsToCheck |= FetchBufferEntryFlagsPredicted;
52 }
53
54 return 0;
55}

155void Fetcher::ClockSendBuffered() {
156 unsigned long i = 0;
157
158 // Skip instructions we already sent.
159 while (this->fetchBuffer[i].flags & FetchBufferEntryFlagsSentToMemory) ++i;
160
161 while (i < this->fetchBufferUsage &&
162 (this->instructionMemory->SendRequest(
163 this->instructionMemoryID, &this->fetchBuffer[i].instruction) ==
164 0)) {
165 this->fetchBuffer[i].flags |= FetchBufferEntryFlagsSentToMemory;
166 ++i;
167 }
168
169 // Same thing for the predictor.
170 if (this->predictor == NULL) return;
171
172 i = 0;
173
174 while (i < this->fetchBufferUsage &&
175 this->fetchBuffer[i].flags & FetchBufferEntryFlagsSentToPredictor)
176 ++i;
177
178 while (i < this->fetchBufferUsage) {
179 PredictorPacket packet;
180 packet.type = PredictorPacketTypeRequestQuery;
181 packet.data.requestQuery = this->fetchBuffer[i].instruction;
182 if (this->predictor->SendRequest(this->predictorID, &packet) != 0)
183 break;
184 this->fetchBuffer[i].flags |= FetchBufferEntryFlagsSentToPredictor;
185 ++i;
186 }
187}

189int Fetcher::ClockCheckPredictor() {
190 if (this->predictor == NULL) return 0;
191
192 PredictorPacket response;
193 unsigned long i = 0;
194 // We depend on the predictor sending the responses in order and, of course,
195 // sending only what we actually asked for.
196 while (this->predictor->ReceiveResponse(this->predictorID, &response) ==
197 0) {
198 assert(this->fetchBuffer[i].instruction.staticInfo ==
199 response.data.response.instruction.staticInfo);
200 this->fetchBuffer[i].flags |= FetchBufferEntryFlagsPredicted;
201 long target =
202 this->fetchBuffer[i].instruction.staticInfo->opcodeAddress +
203 this->fetchBuffer[i].instruction.staticInfo->opcodeSize;
204 // "Redirect" the fetch only if the predictor has an address, otherwise
205 // expect the instruction to be at the next logical PC.
206 if (response.type == PredictorPacketTypeResponseTakeToAddress) {
207 target = response.data.response.target;
208 }
209 // If a missprediction happened.
210 if (target != this->fetchBuffer[i].instruction.nextInstruction) {
211 return 1;
212 }
213 ++i;
214 }
215
216 return 0;
217}

219void Fetcher::ClockUnbuffer() {
220 unsigned long i = 0;
221 while (i < this->fetchBufferUsage &&
222 ((this->fetchBuffer[i].flags & this->flagsToCheck) ==
223 this->flagsToCheck)) {
224 ++i;
225 }
226 this->fetchBufferUsage -= i;
227 if (this->fetchBufferUsage > 0) {
228 memmove(this->fetchBuffer, &this->fetchBuffer[i],
229 sizeof(*this->fetchBuffer) * this->fetchBufferUsage);
230 }
231}

233void Fetcher::ClockRequestFetch() {
234 unsigned long fetchBufferByteUsage = 0;
235 for (unsigned long i = 0; i < this->fetchBufferUsage; ++i) {
236 fetchBufferByteUsage +=
237 this->fetchBuffer[i].instruction.staticInfo->opcodeSize;
238 }
239
240 FetchPacket request;
241 request.request = fetchSize - fetchBufferByteUsage;
242 this->fetch->SendRequest(this->fetchID, &request);
243}

245void Fetcher::ClockFetch() {
246 // We're guaranteed to have space because we asked only enough bytes to fill
247 // the buffer. The engine is guaranteed to send only up until that amount,
248 // the cycle right after we asked.
249 while (this->fetch->ReceiveResponse(
250 this->fetchID,
251 (FetchPacket*)&this->fetchBuffer[this->fetchBufferUsage]) == 0) {
252 this->fetchBuffer[this->fetchBufferUsage].flags = 0;
253 ++this->fetchBufferUsage;
254 ++this->fetchedInstructions;
255 }
256}

258void Fetcher::Clock() {
259 // If paying a misspredict penalty.
260 if (this->currentPenalty > 0) {
261 --this->currentPenalty;
262 return;
263 }
264
265 this->ClockSendBuffered();
266 const int predictionResult = this->ClockCheckPredictor();
267 this->ClockUnbuffer();
268
269 // Don't fetch if a misspredict happened. The fetchClock is set to 0 so when
270 // the missprediction is paid, we start fetching immediatly.
271 if (predictionResult != 0) {
272 ++this->misspredictions;
273 this->currentPenalty = this->misspredictPenalty;
274 this->fetchClock = 0;
275 return;
276 }
277
278 this->ClockFetch();
279
280 if (this->fetchClock % this->fetchInterval == 0) {
281 this->fetchClock = 0;
282 this->ClockRequestFetch();
283 }
284
285 ++this->fetchClock;
286}
