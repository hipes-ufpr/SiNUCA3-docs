132bool StrStartsWithGomp(const char* s) {
133 const char* prefix = "gomp";
134 for (int i = 0; prefix[i] != '\0'; ++i) {
135 if (std::tolower(s[i]) != prefix[i]) {
136 return false;
137 }
138 }
139 return true;
140}

160VOID InitInstrumentation() {
161 if (isInstrumentating) return;
162 PIN_GetLock(&pinLock, PIN_ThreadId());
163 SINUCA3_LOG_PRINTF("Start of tool instrumentation block\n");
164 isInstrumentating = true;
165 wasInstrumented = true;
166 PIN_ReleaseLock(&pinLock);

169VOID StopInstrumentation() {
170 if (!isInstrumentating || KnobForceInstrumentation.Value()) return;
171 PIN_GetLock(&pinLock, PIN_ThreadId());
172 SINUCA3_LOG_PRINTF("End of tool instrumentation block\n");
173 isInstrumentating = false;
174 PIN_ReleaseLock(&pinLock);

216VOID OnThreadStart(THREADID tid, CONTEXT* ctxt, INT32 flags, VOID* v) {
217 PIN_GetLock(&pinLock, tid);
218 SINUCA3_DEBUG_PRINTF("New thread created! N => %d (%s)\n", tid, imageName);
219 staticTrace->IncThreadCount();
220 isThreadAnalysisEnabled.push_back(
221 KnobForceInstrumentation.Value()); // Init with instru. disabled (or
222 // enabled if forced with "-f")
223 dynamicTraces.push_back(
224 new sinucaTracer::DynamicTraceFile(folderPath, imageName, tid));
225 memoryTraces.push_back(
226 new sinucaTracer::MemoryTraceFile(folderPath, imageName, tid));
227 PIN_ReleaseLock(&pinLock);
228}

231 PIN_GetLock(&pinLock, tid);
232 delete dynamicTraces[tid];
233 delete memoryTraces[tid];
234 PIN_ReleaseLock(&pinLock);

237VOID InsertInstrumentionOnMemoryOperations(const INS* ins) {
238 /*
239 * INS_IsStandardMemop() returns false if this instruction has a memory
240 * operand which has unconventional meaning; returns true otherwise
241 */
242 if (!INS_IsStandardMemop(*ins)) {
243 INS_InsertCall(*ins, IPOINT_BEFORE, (AFUNPTR)AppendToMemTraceNonStd,
244 IARG_MULTI_MEMORYACCESS_EA, IARG_END);
245 return;
246 }
247
248 bool isRead = INS_IsMemoryRead(*ins);
249 bool hasRead2 = INS_HasMemoryRead2(*ins);
250 bool isWrite = INS_IsMemoryWrite(*ins);
251
252 if (isRead) {
253 INS_InsertCall(*ins, IPOINT_BEFORE, (AFUNPTR)AppendToMemTraceStd,
254 MEMREAD_EA, MEMREAD_SIZE, IARG_END);
255 }
256 if (hasRead2) {
257 INS_InsertCall(*ins, IPOINT_BEFORE, (AFUNPTR)AppendToMemTraceStd,
258 MEMREAD2_EA, MEMREAD_SIZE, IARG_END);
259 }
260 if (isWrite) {
261 INS_InsertCall(*ins, IPOINT_BEFORE, (AFUNPTR)AppendToMemTraceStd,
262 MEMWRITE_EA, MEMWRITE_SIZE, IARG_END);
263 }
264}

266VOID OnTrace(TRACE trace, VOID* ptr) {
267 if (!isInstrumentating) return;
268
269 THREADID tid = PIN_ThreadId();
270 RTN traceRtn = TRACE_Rtn(trace);
271
272 if (RTN_Valid(traceRtn)) {
273 const char* traceRtnName = RTN_Name(traceRtn).c_str();
274#if DEBUG_PRINT_GOMP_RNT == 1
275 if (StrStartsWithGomp(traceRtnName)) {
276 TRACE_InsertCall(trace, IPOINT_BEFORE, (AFUNPTR)DebugPrintRtnName,
277 IARG_PTR, traceRtnName, IARG_THREAD_ID, IARG_END);
278 }
279#endif
280 /*
281 * This will make every function call from libgomp that have a
282 * PAUSE instruction to be ignored
283 * Still not sure if this is fully correct
284 */
285 for (unsigned int i = 0; i < OMP_ignore.size(); ++i) {
286 if (strcmp(traceRtnName, OMP_ignore[i]) == 0) {
287 // has SPIN_LOCK
288 return;
289 }
290 }
291 }
292
293 PIN_GetLock(&pinLock, tid);
294
295 for (BBL bbl = TRACE_BblHead(trace); BBL_Valid(bbl); bbl = BBL_Next(bbl)) {
296 unsigned int numberInstBBl = BBL_NumIns(bbl);
297 BBL_InsertCall(bbl, IPOINT_ANYWHERE, (AFUNPTR)AppendToDynamicTrace,
298 IARG_UINT32, staticTrace->GetBBlCount(), IARG_UINT32,
299 numberInstBBl, IARG_END);
300
301 staticTrace->IncBBlCount();
302 staticTrace->AppendToBufferNumIns(numberInstBBl);
303 for (INS ins = BBL_InsHead(bbl); INS_Valid(ins); ins = INS_Next(ins)) {
304 staticTrace->PrepareDataINS(&ins);
305 staticTrace->AppendToBufferDataINS();
306 InsertInstrumentionOnMemoryOperations(&ins);
307 staticTrace->IncInstCount();
308 }
309 }
310
311 PIN_ReleaseLock(&pinLock);
312 return;
313}

315VOID OnImageLoad(IMG img, VOID* ptr) {
316 if (IMG_IsMainExecutable(img) == false) return;
317
318 std::string completeImgPath = IMG_Name(img);
319 unsigned long imgPathLen = completeImgPath.size();
320 const char* completeImgPathPtr = completeImgPath.c_str();
321 // As Pin gives us the absolute path for the executable, it always have at
322 // least a '/' (the root of the filesystem).
323 // We need to find the word after the last '/'.
324 int idx = 0;
325 for (int i = imgPathLen - 1; i >= 0; --i) {
326 if (completeImgPathPtr[i] == '/') {
327 idx = i + 1;
328 break;
329 }
330 }
331 unsigned long imageNameLen = imgPathLen - idx + sizeof('\0');
332 imageName = (char*)malloc(imageNameLen); // freed in Fini()
333 memcpy(imageName, completeImgPathPtr + idx, imageNameLen);
334
335 staticTrace = new sinucaTracer::StaticTraceFile(folderPath, imageName);
336 for (SEC sec = IMG_SecHead(img); SEC_Valid(sec); sec = SEC_Next(sec)) {
337 for (RTN rtn = SEC_RtnHead(sec); RTN_Valid(rtn); rtn = RTN_Next(rtn)) {
338 RTN_Open(rtn);
339 const char* name = RTN_Name(rtn).c_str();
340
341 if (strcmp(name, "BeginInstrumentationBlock") == 0) {
342 RTN_InsertCall(rtn, IPOINT_BEFORE, (AFUNPTR)InitInstrumentation,
343 IARG_END);
344 }
345
346 if (strcmp(name, "EndInstrumentationBlock") == 0) {
347 RTN_InsertCall(rtn, IPOINT_BEFORE, (AFUNPTR)StopInstrumentation,
348 IARG_END);
349 }
350
351 if (strcmp(name, "EnableThreadInstrumentation") == 0) {
352 RTN_InsertCall(rtn, IPOINT_BEFORE,
353 (AFUNPTR)EnableInstrumentationInThread,
354 IARG_THREAD_ID, IARG_END);
355 }
356
357 if (strcmp(name, "DisableThreadInstrumentation") == 0) {
358 RTN_InsertCall(rtn, IPOINT_BEFORE,
359 (AFUNPTR)DisableInstrumentationInThread,
360 IARG_THREAD_ID, IARG_END);
361 }
362
363 RTN_Close(rtn);
364 }
365 }

368VOID OnFini(INT32 code, VOID* ptr) {
369 SINUCA3_LOG_PRINTF("End of tool execution\n");
370 SINUCA3_DEBUG_PRINTF("Number of BBLs [%u]\n", staticTrace->GetBBlCount());
371
372 if (imageName != NULL) free(imageName);
373
374 // Close static trace file
375 delete staticTrace;
376
377 if (!wasInstrumented) {
378 SINUCA3_WARNING_PRINTF(
379 "No instrumentation blocks were found in the target program.\n"
380 "As result, no instruction data was recorded in the trace files.\n"
381 "To enable instrumentation, wrap the target code with "
382 "BeginInstrumentationBlock() and EndInstrumentationBlock().\n"
383 "Instrumentation code is only inserted within these blocks, and "
384 "analysis is only executed if the thread has\n"
385 "called EnableThreadInstrumentation().\n"
386 "Use the -f flag to force instrumentation even when no blocks are "
387 "defined.\n"
388 "Use the -f flag to force instrumentation even when no blocks are "
389 "defined.\n");
390 }

393int main(int argc, char* argv[]) {
394 PIN_InitSymbols();
395
396 if (PIN_Init(argc, argv)) {
397 return Usage();
398 }
399
400 folderPath = KnobFolder.Value().c_str();
401 if (access(folderPath, F_OK) != 0) {
402 mkdir(folderPath, S_IRWXU | S_IRWXG | S_IROTH);
403 }
404
405 PIN_InitLock(&pinLock);
406
407 // Init with instru. disabled (or enabled if forced with "-f");
408 if (KnobForceInstrumentation.Value()) {
409 SINUCA3_WARNING_PRINTF(
410 "Force flag (\"-f\") enabled: Instrumentating entire program for "
411 "all created threads.\n");
412 InitInstrumentation();
413 } else {
414 isInstrumentating = false;
415 }
416
417 InitGompHandling();
418
419 IMG_AddInstrumentFunction(OnImageLoad, NULL);
420 TRACE_AddInstrumentFunction(OnTrace, NULL);
421 PIN_AddFiniFunction(OnFini, NULL);
422
423 PIN_AddThreadStartFunction(OnThreadStart, NULL);
424 PIN_AddThreadFiniFunction(OnThreadFini, NULL);
425
426 // Never returns
427 PIN_StartProgram();
428
429 return 0;
