71int EngineDebugComponent::SetConfigParameter(const char* parameter,
72 ConfigValue value) {
73 this->PrintConfigValue(parameter, value);
74 if (strcmp(parameter, "failNow") == 0) {
75 SINUCA3_DEBUG_PRINTF("%p: SetConfigParameter returning failure.\n",
76 this);
77 return 1;
78 }
79 if (strcmp(parameter, "failOnFinish") == 0) {
80 SINUCA3_DEBUG_PRINTF("%p: Will fail on finish.\n", this);
81 this->shallFailOnFinish = true;
82 return 0;
83 }
84 if (strcmp(parameter, "pointerOther") == 0) {
85 this->other =
86 dynamic_cast<EngineDebugComponent*>(value.value.componentReference);
87 if (this->other == NULL) {
88 SINUCA3_DEBUG_PRINTF(
89 "%p: Failed to cast pointerOther to EngineDebugComponent.\n",
90 this);
91 return 1;
92 }
93 this->otherConnectionID = this->other->Connect(4);
94 return 0;
95 }
96 if (strcmp(parameter, "fetch") == 0) {
97 this->fetch = dynamic_cast<Component<FetchPacket>*>(
98 value.value.componentReference);
99 if (this->fetch == NULL) {
100 SINUCA3_DEBUG_PRINTF(
101 "%p: Failed to cast fetch to Component<InstructionPacket>.\n",
102 this);
103 return 1;
104 }
105 this->fetchConnectionID = this->fetch->Connect(0);
106 }
107
108 return 0;

117void EngineDebugComponent::Clock() {
118 SINUCA3_DEBUG_PRINTF("%p: Clock!\n", this);
119
120 if (this->fetch != NULL) {
121 FetchPacket packet;
122 packet.request = 0;
123 SINUCA3_DEBUG_PRINTF("%p: Fetching instruction!\n", this);
124 this->fetch->SendRequest(this->fetchConnectionID, &packet);
125 if (this->fetch->ReceiveResponse(this->fetchConnectionID, &packet) ==
126 0) {
127 SINUCA3_DEBUG_PRINTF("%p: Received instruction %s\n", this,
128 packet.response.staticInfo->opcodeAssembly);
129 }
130 }
131
132 InstructionPacket messageInput = {NULL, DynamicInstructionInfo(), 0};
133 InstructionPacket messageOutput = {NULL, DynamicInstructionInfo(), 0};
134
135 if (this->other) {
136 if (!(this->send)) {
137 messageInput.staticInfo = (const StaticInstructionInfo*)0xcafebabe;
138 SINUCA3_DEBUG_PRINTF("%p: Sending message (%p) to %p.\n", this,
139 messageInput.staticInfo, this->other);
140 this->other->SendRequest(this->otherConnectionID, &messageInput);
141 this->send = true;
142 } else {
143 if (this->other->ReceiveResponse(this->otherConnectionID,
144 &messageOutput) == 0) {
145 SINUCA3_DEBUG_PRINTF("%p: Received response (%p) from %p.\n",
146 this, messageOutput.staticInfo,
147 this->other);
148 this->send = false;
149 } else {
150 SINUCA3_DEBUG_PRINTF("%p: No response from %p.\n", this,
151 this->other);
152 }
153 }
154 } else {
155 for (long i = 0; i < this->GetNumberOfConnections(); ++i) {
156 if (this->ReceiveRequestFromConnection(i, &messageOutput) == 0) {
157 SINUCA3_DEBUG_PRINTF("%p: Received message (%p)\n", this,
158 messageOutput.staticInfo);
159 messageInput.staticInfo = messageOutput.staticInfo + 1;
160 SINUCA3_DEBUG_PRINTF("%p: Sending response (%p)\n", this,
161 messageInput.staticInfo);
162 this->SendResponseToConnection(i, &messageInput);
163 }
164 }
165 }
166}
