37int Engine::SendBufferedAndFetch(int id) {
38 InstructionPacket toSend = this->fetchBuffers[id];
39 const FetchResult r = this->traceReader->Fetch(&this->fetchBuffers[id], id);
40 toSend.nextInstruction = this->fetchBuffers[id].staticInfo->opcodeAddress;
41
42 // This unfortunately drops the packet if the buffer is full. The component
43 // must ensure the buffers never fills.
44 if (this->SendResponseToConnection(id, (FetchPacket*)&toSend) != 0) {
45 SINUCA3_WARNING_PRINTF(
46 "engine: == INSTRUCTION DROP DETECTED == core %d made requests "
47 "with a full buffer, instructions will be dropped.\n",
48 id);
49 }
50
51 if (r == FetchResultEnd) {
52 this->end = true;
53 return 1;
54 } else if (r == FetchResultError) {
55 this->error = true;
56 return 1;
57 }
58
59 ++this->fetchedInstructions;
60
61 return 0;
62}

64void Engine::Fetch(int id, FetchPacket packet) {
65 if (packet.request == 0) {
66 this->SendBufferedAndFetch(id);
67 return;
68 }
69
70 long weight = this->fetchBuffers[id].staticInfo->opcodeSize;
71
72 while (weight < packet.request) {
73 if (this->SendBufferedAndFetch(id)) {
74 return;
75 }
76 weight += this->fetchBuffers[id].staticInfo->opcodeSize;
77 }
78}

80void Engine::Clock() {
81 FetchPacket packet;
82 const int numberOfConnections = this->GetNumberOfConnections();
83
84 for (int i = 0; i < numberOfConnections; ++i) {
85 if (!this->ReceiveRequestFromConnection(i, &packet)) {
86 this->Fetch(i, packet);

105void Engine::PrintTime(time_t start, unsigned long cycle) {
106 const unsigned long remaining = this->traceSize - this->fetchedInstructions;
107
108 SINUCA3_LOG_PRINTF("engine: Heartbeat at cycle %ld.\n", cycle);
109 SINUCA3_LOG_PRINTF("engine: Remaining instructions: %ld.\n", remaining - 1);
110
111 const time_t now = time(NULL);
112 const time_t estimatedEnd = (traceSize * (now - start) / remaining) + start;
113 SINUCA3_LOG_PRINTF("engine: Estimated simulation end: %s",
114 ctime(&estimatedEnd));
115}

117int Engine::SetupSimulation(TraceReader* traceReader) {
118 this->traceReader = traceReader;
119 this->numberOfFetchers = this->GetNumberOfConnections();
120 this->fetchBuffers = new InstructionPacket[this->numberOfFetchers];
121
122 // Bufferize the first instruction of each core.
123 for (long i = 0; i < this->numberOfFetchers; ++i) {
124 if (this->traceReader->Fetch(&this->fetchBuffers[i], i) !=
125 FetchResultOk) {
126 return 1;
127 }
128 ++this->fetchedInstructions;
129 }
130
131 return 0;
132}

134int Engine::Simulate(TraceReader* traceReader) {
135 if (this->SetupSimulation(traceReader)) {
136 return 1;
137 }
138
139 this->traceSize = this->GetTraceSize();
140
141 const time_t start = time(NULL);
142
143 SINUCA3_LOG_PRINTF("engine: Simulation started at %s", ctime(&start));
144 SINUCA3_LOG_PRINTF("engine: Total instructions: %ld.\n", this->traceSize);
145
146 while (!this->end && !this->error) {
147 if ((this->totalCycles + 1) % (1 << 8) == 0)
148 this->PrintTime(start, this->totalCycles + 1);
149
150 for (long i = 0; i < this->numberOfComponents; ++i)
151 this->components[i]->Clock();
152
153 for (long i = 0; i < this->numberOfComponents; ++i)
154 this->components[i]->PosClock();
155
156 ++this->totalCycles;
157 }
158
159 const time_t end = time(NULL);
160 SINUCA3_LOG_PRINTF("engine: Simulation ended at %s", ctime(&end));
161 SINUCA3_LOG_PRINTF("=== SIMULATION STATISTICS ===\n");
162
163 if (this->error) {
164 SINUCA3_ERROR_PRINTF(
165 "Simulation ended due to error in trace fetching!\n");
166 }
167
168 for (long i = 0; i < this->numberOfComponents; ++i) {
169 this->components[i]->PrintStatistics();
170 }
171
172 return this->error;
173}
