109template <typename Type>
110int DelayQueue<Type>::Enqueue(Input* elem) {
111 if (this->IsFull()) {
112 return 1;
113 }
114 elem->SetRemoval(this->cyclesClock + this->delay);
115 if (this->IsEmpty()) {
116 this->queueFirst = *elem;
117 } else {
118 this->delayBuffer.Enqueue(elem);
119 }
120 this->occupation++;
121 return 0;
122}
123
124template <typename Type>
125int DelayQueue<Type>::Dequeue(Input* elem) {
126 if (this->IsEmpty()) {
127 return 1;
128 }
129 if (this->queueFirst.removeAt > this->cyclesClock) {
130 return 1; // Not yet time to remove
131 }
132 *elem = this->queueFirst;
133 if (this->occupation > 1) {
134 this->delayBuffer.Dequeue(&this->queueFirst);
135 }
136 this->occupation--;
137 return 0;
138}

179template <typename Type>
180int DelayQueue<Type>::FinishSetup() {
181 if (this->sendTo == NULL) {
182 SINUCA3_ERROR_PRINTF("DelayQueue sendTo param not set.\n");
183 return 1;
184 }
185 if (this->throughput == 0) {
186 SINUCA3_ERROR_PRINTF("DelayQueue throughput param not set.\n");
187 return 1;
188 }
189
190 this->sendToId = this->sendTo->Connect(this->throughput);
191 if (this->UseDelayBuffer()) {
192 this->SetDelayBufferSize();
193 this->delayBuffer.Allocate(this->delayBufferSize, sizeof(Input));
194 }
195 return 0;
196}

198template <typename Type>
199void DelayQueue<Type>::Clock() {
200 this->cyclesClock++;
201
202 Input input;
203 if (this->UseDelayBuffer()) {
204 while (this->Dequeue(&input) == 0) {
205 this->sendTo->SendRequest(sendToId, &input.elem);
206 }
207 }
208 long totalConnections = this->GetNumberOfConnections();
209 for (long i = 0; i < totalConnections; i++) {
210 while (this->ReceiveRequestFromConnection(i, &input.elem) == 0) {
211 if (this->UseDelayBuffer()) {
212 if (this->Enqueue(&input)) return;
213 } else {
214 if (this->sendTo->SendRequest(sendToId, &input.elem)) return;
215 }
216 }
217 }
218}
