59bool CircularBuffer::Enqueue(void* elementInput) {
60 if (this->IsFull()) {
61 if (this->maxBufferSize != 0) {
62 return 1;
63 }
64
65 void* newBuffer =
66 (void*)malloc(this->bufferSize * 2 * this->elementSize);
67 if (this->startOfBuffer >= this->endOfBuffer) {
68 long offset = this->bufferSize - this->startOfBuffer;
69 memcpy(newBuffer, &((char*)this->buffer)[this->startOfBuffer],
70 offset);
71 memcpy(&((char*)newBuffer)[offset], this->buffer,
72 this->bufferSize - offset);
73 this->startOfBuffer = 0;
74 this->endOfBuffer = this->bufferSize - 1;
75 } else {
76 memcpy(newBuffer, this->buffer, this->bufferSize * elementSize);
77 }
78
79 this->bufferSize *= 2;
80 free(this->buffer);
81 this->buffer = newBuffer;
82 }
83
84 /*
85 * Target stores the memory address where the element should be
86 * inserted, based on pointer arithmetic. After its definition, memcpy
87 * stores the element in the most recent position in the buffer.
88 */
89 void* memoryAddress = static_cast<char*>(this->buffer) +
90 (this->endOfBuffer * this->elementSize);
91
92 memcpy(memoryAddress, elementInput, this->elementSize);
93 ++this->occupation;
94 ++this->endOfBuffer;
95
96 if (this->endOfBuffer == this->bufferSize) {
97 this->endOfBuffer = 0;
98 }
99
100 return 0;
101}

103bool CircularBuffer::Dequeue(void* elementOutput) {
104 if (!this->IsEmpty()) {
105 /*
106 * Element stores the memory address of the oldest element in the Buffer
107 * (the one that should be removed). Although there is no need to clear
108 * the space of this element, the buffer limits are readjusted to avoid
109 * unauthorized access.
110 */
111 void* memoryAddress = static_cast<char*>(this->buffer) +
112 (this->startOfBuffer * this->elementSize);
113
114 memcpy(elementOutput, memoryAddress, this->elementSize);
115 --this->occupation;
116 ++this->startOfBuffer;
117
118 if (this->startOfBuffer == this->bufferSize) {
119 this->startOfBuffer = 0;
120 }
121
122 return 0;
123 }
124
125 memset(elementOutput, 0, this->elementSize);
126
127 return 1;
128}
