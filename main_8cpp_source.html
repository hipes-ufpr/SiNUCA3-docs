97int main(int argc, char* const argv[]) {
98 const char* traceReaderName = "sinuca3";
99 const char* rootConfigFile = NULL;
100 const char* traceDir = ".";
101 const char* traceFileName = NULL;
102 char nextOpt;
103
104 // When compiling debug mode, enable our testing facilities.
105#ifdef NDEBUG
106#define SINUCA3_SWITCHES "lc:t:d:T:"
107#else
108#define SINUCA3_SWITCHES "r:lc:t:d:T:"
109 const char* testToRun = NULL;
110#endif
111
112 while ((nextOpt = getopt(argc, argv, SINUCA3_SWITCHES)) != -1) {
113 switch (nextOpt) {
114 // When compiling debug mode, enable our testing facilities.
115#ifndef NDEBUG
116 case 'r':
117 testToRun = optarg;
118 break;
119#endif
120 case 'c':
121 rootConfigFile = optarg;
122 break;
123 case 't':
124 traceFileName = optarg;
125 break;
126 case 'd':
127 traceDir = optarg;
128 break;
129 case 'T':
130 traceReaderName = optarg;
131 break;
132 case 'l':
133 license();
134 return 0;
135 case 'h':
136 usage();
137 return 0;
138 }
139 }
140
141 // When compiling debug mode and there's a test to run, run it.
142#ifndef NDEBUG
143 if (testToRun != NULL) {
144 int ret = Test(testToRun);
145 if (ret < 0) {
146 SINUCA3_LOG_PRINTF("No such test: %s\n", testToRun);
147 } else if (ret > 0) {
148 SINUCA3_LOG_PRINTF("Test failed with code %d.\n", ret);
149 }
150 return ret;
151 }
152#endif
153
154 if (rootConfigFile == NULL) {
155 usage();
156 return 1;
157 }
158 if (traceFileName == NULL) {
159 usage();
160 return 1;
161 }
162
163 EngineBuilder builder;
164 Engine* engine = builder.Instantiate(rootConfigFile);
165 if (engine == NULL) return 1;
166
167 TraceReader* traceReader = AllocTraceReader(traceReaderName);
168 if (traceReader == NULL) {
169 SINUCA3_ERROR_PRINTF("The trace reader %s does not exist.",
170 traceReaderName);
171 return 1;
172 }
173 if (traceReader->OpenTrace(traceFileName, traceDir)) return 1;
174
175 engine->Simulate(traceReader);
176 delete engine;
177 delete traceReader;
178
179 return 0;
180}
