50sinucaTracer::DynamicTraceFile::~DynamicTraceFile() {
51 SINUCA3_DEBUG_PRINTF("Last DynamicTraceFile flush\n");
52 if (this->tf.offsetInBytes > 0) {
53 this->FlushBuffer();
54 }
55
56 rewind(this->tf.file);
57 fwrite(&this->totalExecInst, sizeof(this->totalExecInst), 1, this->tf.file);
58 SINUCA3_DEBUG_PRINTF("totalExecInst [%lu]\n", this->totalExecInst);
59}
